    "```octave\n",
    "'''Algorithm to find the two largest numbers in an array a of n >= 2 elements'''\n",
    "Input: a, n\n",
    "Set i = 0 # index of largest\n",
    "Set j = 1 # index of second largest\n",
    "If a[0] < a[1] then\n",
    "    Set i = 1\n",
    "    Set j = 0\n",
    "Set c = 2 # counter\n",
    "While c < n\n",
    "    If a[c] > a[j] then\n",
    "        Set j = c\n",
    "    If a[c] > a[i] then\n",
    "        Set j = i\n",
    "        Set i = c\n",
    "    Set c = c + 1\n",
